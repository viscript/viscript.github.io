—DDL—

create table cats(
id int not null auto_increment,
pid int not null default ‘0’,
name varchar(60) not null default ‘’,
desn text not null default ‘’,
primary key(id),
index name(name,pid)
);

create table products(
id int not null auto_increment,
cid int not null default ‘0’,
name varchar(60) not null default ‘’,
price double(7,2) not null default ‘0.00’,
num int not null default ‘0’,
desn text,
ptime int not null default ‘0’,
primary key(id),
key pname(name, price)
);

select name from tab1,tab2; //如果name是两个表共有的，会出错。

select tab1.name,tab2.name from tab1,tab2;
select t1.name 1name,t2.name 2name from tab1 t1,tab2,t2; //分别对表和元素分别别名
select t1.name 1name,t2.name 2name from tab1 t1,tab2,t2 where t1.id=t2.id; //将t2的id关联到了t1的id，解决了笛卡尔乘积的问题。
